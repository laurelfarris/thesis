%% Deluxetable syntax
%\onecolumngrid  % switch from two columns to one to make table span entire width of page
\begin{deluxetable}{l l l}
    %\tabletypesize{font size command} --> \small (11pt), \footnotesize (10pt), or \scriptsize (8pt).
    %\rotate % --> landscape
    \tablewidth{\textwidth} % default = \pagewidth, natural width = 0pt
    \tablenum{\thesection\thetable}
    \tablecolumns{num}
    \tablecaption{
        Table caption text goes here.
        \label{key}}
    \tablehead{\colhead{\nodata} & \colhead{heading1} & \colhead{heading2}}
    \startdata
    string & num1 & num2\\%[dimen]
        % dimen --> dimension of vertical space;
        % may be specified in any units that are legitimate in LATEX.
    \enddata
\end{deluxetable}%--
